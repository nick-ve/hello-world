178void IceXtalk::Exec(Option_t* opt)
179{
185
186 TString name=opt;
187 NcJob* parent=(NcJob*)(gROOT->GetListOfTasks()->FindObject(name.Data()));
188
189 if (!parent) return;
190
191 IceEvent* evt=(IceEvent*)parent->GetObject("IceEvent");
192 if (!evt) return;
193
194 // Only process accepted events
195 NcDevice* seldev=(NcDevice*)evt->GetDevice("NcEventSelector");
196 if (seldev)
197 {
198 if (seldev->GetSignal("Select") < 0.1) return;
199 }
200
201 NcSignal* daq=(NcSignal*)evt->GetDevice("Daq");
202 Int_t mudaq=0;
203 if (daq) mudaq=int(daq->GetSignal("Muon"));
204
205 // This cross talk correction processor is only for MuDaq data
206 if (!mudaq) return;
207
208 if (!fOmdb && fCalfile)
209 {
210 fOmdb=(NcObjMatrix*)fCalfile->Get("MuDaq-OMDBASE");
211 // Next statement for compatibility with old calibration file format
212 if (!fOmdb) fOmdb=(NcObjMatrix*)fCalfile->Get("Cal-OMDBASE");
213 }
214
215 if (!fOmdb) return;
216
217 // Storage of the used parameters in the IceXtalk device
218 NcDevice params;
219 params.SetNameTitle("IceXtalk","IceXtalk processor parameters");
220 params.SetSlotName("Pmin",1);
221 params.SetSlotName("Pe",2);
222
223 params.SetSignal(fPmin,1);
224 params.SetSignal(fPe,2);
225
226 evt->AddDevice(params);
227
228 // All Amanda OMs with a signal
229 TObjArray* mods=evt->GetDevices("IceAOM");
230 if (!mods) return;
231 Int_t nmods=mods->GetEntries();
232 if (!nmods) return;
233
234 TObjArray xhits; // Array with pointers to Xtalk hits to be corrected
235
236 IceAOM* omt=0; // Transmitter OM
237 IceAOM* omr=0; // Receiver OM
238 TF1* fxtalk=0; // The Xtalk probability function
239 Int_t idtrans=0;
240 Int_t idrec=0;
241 Float_t adct=0,let=0;
242 Float_t p=0;
243 Float_t adcr=0,ler=0;
244 Int_t ibad=0;
245 Float_t cpar=0,bpar=0,dlemin=0,dlemax=0,test=0;
246 Float_t dle=0;
247 Float_t sigcor=0;
248 for (Int_t imod=0; imod<nmods; imod++)
249 {
250 omt=(IceAOM*)mods->At(imod);
251 if (!omt) continue;
252 idtrans=omt->GetUniqueID();
253
254 // Also take bad transmitter modules into account
255 ibad=omt->GetDeadValue("ADC");
256 if (ibad) omt->SetAlive("ADC");
257
258 // Check for corresponding receiver modules
259 for (Int_t jmod=0; jmod<nmods; jmod++)
260 {
261 // No Xtalk from a module to itself
262 if (jmod==imod) continue;
263
264 omr=(IceAOM*)mods->At(jmod);
265 if (!omr) continue;
266 idrec=omr->GetUniqueID();
267
268 fxtalk=(TF1*)fOmdb->GetObject(idtrans,idrec+1);
269 if (!fxtalk) continue;
270
271 // Determine Xtalk probability for each transmitter hit
272 for (Int_t ithit=1; ithit<=omt->GetNhits(); ithit++)
273 {
274 NcSignal* st=omt->GetHit(ithit);
275 if (!st) continue;
276
277 // Eliminate possible previous Xtalk correction
278 sigcor=st->GetOffset("ADC");
279 st->AddSignal(sigcor,"ADC");
280 st->ResetOffset("ADC");
281 adct=st->GetSignal("ADC",-4); // Get uncalibrated amplitude
282 dlemin=fxtalk->GetParameter("dLE-min");
283 dlemax=fxtalk->GetParameter("dLE-max");
284 // Protection against overflow in Xtalk prob. function
285 cpar=fxtalk->GetParameter("C");
286 bpar=fxtalk->GetParameter("B");
287 test=(adct-cpar)/bpar;
288 if (test>100)
289 {
290 p=1;
291 }
292 else if (test<-100)
293 {
294 p=0;
295 }
296 else
297 {
298 p=fxtalk->Eval(adct);
299 }
300 if (p<fPmin) continue;
301
302 // Xtalk flagging for each receiver hit
303 for (Int_t irhit=1; irhit<=omr->GetNhits(); irhit++)
304 {
305 NcSignal* sr=omr->GetHit(irhit);
306 if (!sr) continue;
307
308 // Check calibrated LE time differences
309 if (!ibad)
310 {
311 let=st->GetSignal("LE",4);
312 ler=sr->GetSignal("LE",4);
313 dle=ler-let;
314 if (dle<dlemin || dle>dlemax) continue;
315 }
316 // Register this receiver hit to be corrected
317 xhits.Add(sr);
318 }
319 } // end of transmitter hit loop
320 } // end of receiver OM loop
321
322 // Restore the original transmitter dead flag value
323 if (ibad) omt->SetDead("ADC");
324 } // end of transmitter OM loop
325
326 // Perform the Xtalk signal correction to the registered receiver hits
327 // The correction value is stored in the signal data as an offset value
328 for (Int_t ix=0; ix<xhits.GetEntries(); ix++)
329 {
330 NcSignal* sx=(NcSignal*)xhits.At(ix);
331 if (!sx) continue;
332
333 // Eliminate possible previous Xtalk correction
334 sigcor=sx->GetOffset("ADC");
335 adcr=sx->GetSignal("ADC")+sigcor;
336 sigcor=fPe;
337 // If stored ADC data is un-calibrated, convert fPe to raw ADC value
338 NcSignal* parent=(NcSignal*)sx->GetDevice();
339 if (parent)
340 {
341 if (parent->GetCalFunction("ADC"));
342 {
343 idrec=parent->GetUniqueID();
344 omr=(IceAOM*)fOmdb->GetObject(idrec,1);
345 TF1* fdecal=0;
346 if (omr) fdecal=omr->GetDecalFunction("ADC");
347 if (fdecal) sigcor=fdecal->Eval(fPe);
348 }
349 }
350 adcr=adcr-sigcor;
351 sx->SetSignal(adcr,"ADC");
352 sx->SetOffset(sigcor,"ADC");
353 }
354}
