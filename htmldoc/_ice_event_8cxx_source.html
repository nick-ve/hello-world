342Int_t IceEvent::GetNstrings(TString classname)
343{
349
350 // Fetch all fired modules of the specified class for this event
351 TObjArray* mods=GetDevices(classname);
352 if (!mods) return 0;
353 Int_t nmods=mods->GetEntries();
354 if (!nmods) return 0;
355
356 // Check for the string ids of the good fired modules
357 if (!fStrings) fStrings=new TArrayI(200);
358 fStrings->Reset();
359 Int_t idxmax=0;
360 Int_t nstrings=0;
361 Int_t jstring=0;
362 Int_t match=0;
363 for (Int_t imod=0; imod<nmods; imod++)
364 {
365 IceGOM* omx=(IceGOM*)mods->At(imod);
366 if (!omx) continue;
367 if (omx->GetDeadValue("ADC") || omx->GetDeadValue("LE") || omx->GetDeadValue("TOT")) continue;
368
369 // Update the number of fired strings
370 jstring=omx->GetString();
371 match=0;
372 idxmax=nstrings;
373 for (Int_t idx=0; idx<idxmax; idx++)
374 {
375 if (jstring==fStrings->At(idx))
376 {
377 match=1;
378 break;
379 }
380 }
381 if (!match) // String number was not present in array
382 {
383 nstrings++;
384 fStrings->AddAt(jstring,nstrings-1);
385 }
386 }
387
388 return nstrings;

535Int_t IceEvent::GetStringMax(TString classname,Int_t* id,Float_t* x,Float_t* y)
536{
550
551 // Reset the values of the return arguments
552 if (id) *id=0;
553 if (x) *x=0;
554 if (y) *y=0;
555
556 // Fetch all fired modules of the specified class for this event
557 TObjArray* mods=GetDevices(classname);
558 if (!mods) return 0;
559 Int_t nmods=mods->GetEntries();
560 if (!nmods) return 0;
561
562 // Array for the string ids of the good fired modules
563 if (!fStrings) fStrings=new TArrayI(200);
564 fStrings->Reset();
565
566 Int_t ncount=0;
567 Int_t nmax=0;
568 Int_t idmax=0;
569 Float_t xmax=0;
570 Float_t ymax=0;
571 Float_t scale=0;
572 Int_t jstring=0;
573 for (Int_t imod=0; imod<nmods; imod++)
574 {
575 IceGOM* omx=(IceGOM*)mods->At(imod);
576 if (!omx) continue;
577 if (omx->GetDeadValue("ADC") || omx->GetDeadValue("LE") || omx->GetDeadValue("TOT")) continue;
578
579 // Update the count for the corresponding string
580 jstring=omx->GetString();
581 ncount=fStrings->At(jstring);
582 ncount++;
583 fStrings->AddAt(ncount,jstring);
584 if (ncount>nmax)
585 {
586 idmax=jstring;
587 scale=omx->GetUnitScale();
588 xmax=float(omx->GetX(1,"car"))*scale;
589 ymax=float(omx->GetX(2,"car"))*scale;
590 nmax=ncount;
591 fStrings->AddAt(nmax,0);
592 }
593 }
594
595 if (id) *id=idmax;
596 if (x) *x=xmax;
597 if (y) *y=ymax;
598
599 return nmax;
600}

602Float_t IceEvent::GetTriggerTime(TString trigname,TObjArray* arr,Int_t slc,TArrayF* peaks) const

633
634 NcSample times;
635 times.SetStoreMode(1);
636 TString tname;
637 Float_t ttime=0;
638 NcSignal* sx=0;
639 IceGOM* omx=0;
640 TObjArray* oms=0;
641 TObjArray* hits=0;
642 Float_t time=0;
643 Float_t tfirst=999999;
644 Float_t tlast=-999999;
645 Float_t tlow=0;
646 Float_t tup=0;
647 Float_t dtbin=500; // Bin size in ns for peak search histo
648 Int_t nbins=0;
649
650 // Detect from inheritance whether "arr" contains optical modules or hits
651 if (arr)
652 {
653 TObject* obj=arr->At(0);
654 if (obj)
655 {
656 if (obj->InheritsFrom("IceGOM"))
657 {
658 oms=arr;
659 }
660 else
661 {
662 hits=arr;
663 }
664 }
665 }
666
667 if (!arr) // Trigger time derived from trigger data
668 {
669 NcDevice* tdev=(NcDevice*)GetDevice("Trigger");
670 if (!tdev) return 0;
671
672 if (trigname=="Average") // Take median triggertime
673 {
674 for (Int_t itrig=1; itrig<=tdev->GetNhits(); itrig++)
675 {
676 NcSignal* trig=tdev->GetHit(itrig);
677 ttime=0;
678 if (trig)
679 {
680 // Don't take the GLOBAL triggers into account.
681 // They are just there for administrative purposes.
682 tname=trig->GetName();
683 if (tname.Contains("GLOBAL")) continue;
684
685 ttime=trig->GetSignal("trig_pulse_le");
686 times.Enter(ttime);
687 }
688 }
689 ttime=times.GetMedian(1);
690 }
691 else // Take triggertime from the specified trigger
692 {
693 NcSignal* trig=tdev->GetHit(trigname);
694 if (trig) ttime=trig->GetSignal("trig_pulse_le");
695 }
696 }
697
698 if (oms) // Trigger time from array of optical modules via (median of) recorded hit times
699 {
700 for (Int_t iom=0; iom<oms->GetEntries(); iom++)
701 {
702 omx=(IceGOM*)oms->At(iom);
703 if (!omx) continue;
704 for (Int_t ih=1; ih<=omx->GetNhits(); ih++)
705 {
706 sx=omx->GetHit(ih);
707 if (!sx) continue;
708 if (sx->GetDeadValue("LE")) continue;
709
710 ttime=sx->GetSignal("LE",7);
711 if (ttime<tfirst) tfirst=ttime;
712 if (ttime>tlast) tlast=ttime;
713
714 if (!slc && sx->GetSignal("SLC")>0.5) continue;
715
716 times.Enter(ttime);
717 }
718 }
719 ttime=times.GetMedian(1);
720 }
721
722 if (hits) // Trigger time from array of hits via (median of) recorded hit times
723 {
724 for (Int_t ih=0; ih<hits->GetEntries(); ih++)
725 {
726 sx=(NcSignal*)hits->At(ih);
727 if (!sx) continue;
728 if (sx->GetDeadValue("LE")) continue;
729
730 ttime=sx->GetSignal("LE",7);
731 if (ttime<tfirst) tfirst=ttime;
732 if (ttime>tlast) tlast=ttime;
733
734 if (!slc && sx->GetSignal("SLC")>0.5) continue;
735
736 times.Enter(ttime);
737 }
738 ttime=times.GetMedian(1);
739 }
740
741 if (peaks)
742 {
743 peaks->Set(1);
744 peaks->AddAt(ttime,0);
745 }
746
747 Int_t ntimes=times.GetN();
748 if (peaks && tlast>tfirst && ntimes>1)
749 {
750 tlow=tfirst-dtbin;
751 tup=tlast+dtbin;
752 nbins=int((tup-tlow)/dtbin);
753 TH1F thist("thist","hit times",nbins,tlow,tup);
754 for (Int_t it=1; it<=ntimes; it++)
755 {
756 time=times.GetEntry(it,1);
757 thist.Fill(time);
758 }
759// Next line modified because of an incompatible change in ROOT6
760//@@@ TSpectrum spec;
761 NcSpectrum spec;
762 spec.SetDeconIterations(100);
763 Int_t oldlevel=gErrorIgnoreLevel;
764 gErrorIgnoreLevel=kFatal; // Suppress all (TSpectrum c.q. NcSpectrum) error and warning messages
765 Int_t npeaks=spec.Search(&thist,1,"goff",0.3);
766 gErrorIgnoreLevel=oldlevel; // Re-activate previous info level
767 if (npeaks)
768 {
769 peaks->Set(npeaks);
770 Float_t* specarr=spec.GetPositionX();
771 for (Int_t ip=0; ip<npeaks; ip++)
772 {
773 peaks->AddAt(specarr[ip],ip);
774 }
775 }
776 }
777
778 return ttime;
779}
