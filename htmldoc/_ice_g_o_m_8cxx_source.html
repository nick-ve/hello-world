294Double_t IceGOM::GetTimeResidual(NcEvent* evt,NcTrack* t,NcSignal* s,TString name,Int_t mode,Int_t vgroup) const
295{
323
324 Double_t tres=-99999;
325
326 if (!evt || !t || !s) return tres;
327
328 IceGOM* om=(IceGOM*)s->GetDevice();
329 if (!om) return tres;
330
331 // No implementation (yet) for IceTOP signals
332 if (om->InheritsFrom("IceTDOM")) return tres;
333
334 Nc3Vector p=t->Get3Momentum();
335 if (!p.HasVector() || !p.GetNorm()) return tres;
336
337 const Float_t pi=acos(-1.);
338 const Float_t c=0.299792458; // Light speed in vacuum in meters per ns
339 const Float_t npice=1.31768387; // Phase refractive index (c/v_phase) of ice
340 const Float_t ngice=1.35075806; // Group refractive index (c/v_group) of ice
341 const Float_t thetac=acos(1./npice); // Cherenkov angle (in radians)
342
343 // Angular reduction of complement of thetac due to v_phase and v_group difference
344 Float_t alphac=0;
345 if (vgroup) alphac=atan((1.-npice/ngice)/sqrt(npice*npice-1.));
346
347 // The track reference point
348 NcPosition* r0=t->GetReferencePoint();
349 if (!r0) return tres;
350
351 // Absolute (UT) time stamp of the track reference point
352 NcTimestamp* tt0=r0->GetTimestamp();
353 if (!tt0) return tres;
354
355 // Time stamp of the track relative to the event time stamp
356 Float_t t0=evt->GetDifference(tt0,"ns");
357
358 NcPosition rhit=om->GetPosition();
359 Float_t d=t->GetDistance(rhit);
360 Nc3Vector r12=rhit-(*r0);
361 Float_t hproj=p.Dot(r12)/p.GetNorm();
362 Float_t dist=fabs(hproj)+d/tan(pi/2.-thetac-alphac);
363 if (hproj<0) dist=-dist;
364 Float_t tgeo=t0+dist/c; // The predicted geometrical hit time
365 Float_t thit=s->GetSignal("LE",7); // Hit time relative to the event time stamp
366
367 tres=thit-tgeo;
368 return tres;
369}
